true_labels = [1, 2, 0, 1, 2, 1, 0, 2, 1, 2, 2, 0]
predicted_labels = [1, 2, 0, 1, 2, 2, 2, 1, 0, 1, 1, 0]
class_dict = {
0 : "Class A",
1 : "Class B",
2 : "Class C",
}

# Metrics for classification
from sklearn.metrics import classification_report, confusion_matrix,
accuracy_score
import seaborn as sb
import matplotlib.pyplot as plt
print("Accuracy :", accuracy_score(true_labels, predicted_labels)* 100)
print("\n")
print("Classification Report :", classification_report(true_labels,
predicted_labels))
cm = confusion_matrix([class_dict[label] for label in true_labels],
[class_dict[label] for label in predicted_labels], labels=["Class A"])
sb.heatmap(cm, annot = True, cmap = "Blues")
plt.show()

cm = confusion_matrix([class_dict[label] for label in true_labels],
[class_dict[label] for label in predicted_labels], labels=["Class B"])
sb.heatmap(cm, annot = True, cmap = "Blues")
plt.show()
cm = confusion_matrix([class_dict[label] for label in true_labels],
[class_dict[label] for label in predicted_labels], labels=["Class C"])
sb.heatmap(cm, annot = True, cmap = "Blues")
plt.show()
